{{/* video_transcript_links.html Resolves transcript data into a slice of transcript link dicts. Source: video_transcript_resources (resolved by full_metadata to array-of-objects). Input: dict with: - "context" : page context (for resource_url resolution) - "transcriptFile" : video_transcript_resources, an array of {"file": ..., "language": ...} Returns: slice of dicts with keys: "href", "language", "locale", "label" */}} {{- $context := .context -}} {{- $transcriptFile := .transcriptFile -}} {{- $links := slice -}} {{- if $transcriptFile -}} {{/* video_transcript_resources is always an array of {file, language, locale?} objects */}} {{/* First pass: check whether any English track exists */}} {{- $hasEnglish := false -}} {{- range $transcriptFile -}} {{- if eq (.language | default "en") "en" -}} {{- $hasEnglish = true -}} {{- end -}} {{- end -}} {{/* Second pass: build links; first English (or first overall) is the default */}} {{- $firstEnglishDone := false -}} {{- $firstDone := false -}} {{- range $transcriptFile -}} {{- $file := .file -}} {{- $lang := .language | default "en" -}} {{- $locale := .locale | default "" -}} {{- $href := partial "resource_url.html" (dict "context" $context "url" $file) -}} {{- $baseLabel := partial "human_language.html" (dict "language" $lang "locale" $locale) -}} {{- $isDefault := false -}} {{- if $hasEnglish -}} {{- if and (eq $lang "en") (not $firstEnglishDone) -}} {{- $isDefault = true -}} {{- $firstEnglishDone = true -}} {{- end -}} {{- else if not $firstDone -}} {{- $isDefault = true -}} {{- $firstDone = true -}} {{- end -}} {{- $label := $baseLabel -}} {{- if $isDefault -}} {{- $label = printf "%s (Default)" $baseLabel -}} {{- end -}} {{- $links = $links | append (dict "href" $href "language" $lang "locale" $locale "label" $label "isDefault" $isDefault) -}} {{- end -}} {{- end -}} {{- return $links -}}
